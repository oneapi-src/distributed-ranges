function(add_mhp_example example_name)
  add_executable(${example_name} ${example_name}.cpp)
  target_link_libraries(
          ${example_name}
          cxxopts
          DR::mpi)
  add_mpi_test(${example_name} ${example_name} 2)
endfunction()
