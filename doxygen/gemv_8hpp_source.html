15template <lib::distributed_range C, typename T, typename I,
16 lib::distributed_range B>
17void gemv(C &&c, shp::sparse_matrix<T, I> &a, B &&b) {
18 assert(c.size() == b.size());
19 assert(a.shape()[1] == b.size());
20 assert(a.grid_shape()[0] == c.segments().size());
21 assert(a.grid_shape()[1] == 1);
22
23 auto &&devices = shp::devices();
24
25 using b_scalar_type = rng::range_value_t<B>;
26
27 using local_vector_type =
28 shp::device_vector<b_scalar_type, shp::device_allocator<b_scalar_type>>;
29
30 std::vector<local_vector_type> local_b;
31 std::vector<sycl::event> copy_events;
32 std::vector<sycl::event> comp_events;
33
34 for (std::size_t i = 0; i < devices.size(); i++) {
35 shp::device_allocator<T> allocator(shp::context(), devices[i]);
36 local_b.push_back(local_vector_type(b.size(), allocator, i));
37 }
38
39 for (auto &&l_b : local_b) {
40 auto event =
41 shp::copy_async(b.begin(), b.end(), lib::ranges::local(l_b.begin()));
42 copy_events.push_back(event);
43 }
44
45 for (size_t i = 0; i < a.grid_shape()[0]; i++) {
46 auto a_tile = a.tile({i, 0});
47
48 auto a_iter = a_tile.begin();
49 auto b_iter = lib::ranges::local(local_b[i].begin());
50 auto c_iter = lib::ranges::local(c.segments()[i].begin());
51
52 auto device = devices[a_tile.rank()];
53 sycl::queue q(shp::context(), device);
54
55 auto event = q.submit([&](auto &&h) {
56 h.depends_on(copy_events[i]);
57 h.parallel_for(a_tile.size(), [=](auto idx) {
58 auto &&[index, a_v] = *(a_iter + idx);
59 auto &&[i, k] = index;
60 auto &&b_v = *(b_iter + k);
61 auto &&c_v = *(c_iter + i);
62 c_v += a_v * b_v;
63 });
64 });
65 comp_events.push_back(event);
66 }
67
68 __detail::wait(comp_events);
69}

71template <lib::distributed_range C, typename T, typename I,
72 lib::distributed_range B>
73void gemv_rows(C &&c, shp::sparse_matrix<T, I> &a, B &&b) {
74 assert(c.size() == b.size());
75 assert(a.shape()[1] == b.size());
76 assert(a.grid_shape()[0] == c.segments().size());
77 assert(a.grid_shape()[1] == 1);
78
79 auto &&devices = shp::devices();
80
81 using b_scalar_type = rng::range_value_t<B>;
82
83 using local_vector_type =
84 shp::device_vector<b_scalar_type, shp::device_allocator<b_scalar_type>>;
85
86 std::vector<local_vector_type> local_b;
87 std::vector<sycl::event> copy_events;
88 std::vector<sycl::event> comp_events;
89
90 for (std::size_t i = 0; i < devices.size(); i++) {
91 shp::device_allocator<T> allocator(shp::context(), devices[i]);
92 local_b.push_back(local_vector_type(b.size(), allocator, i));
93 }
94
95 for (auto &&l_b : local_b) {
96 auto event =
97 shp::copy_async(b.begin(), b.end(), lib::ranges::local(l_b.begin()));
98 copy_events.push_back(event);
99 }
100
101 for (size_t i = 0; i < a.grid_shape()[0]; i++) {
102 auto a_tile = a.tile({i, 0});
103
104 auto a_iter = a_tile.begin();
105 auto b_iter = lib::ranges::local(local_b[i].begin());
106 auto c_iter = lib::ranges::local(c.segments()[i].begin());
107
108 auto device = devices[a_tile.rank()];
109 sycl::queue q(shp::context(), device);
110
111 std::size_t wg = 32;
112
113 auto event = q.submit([&](auto &&h) {
114 h.depends_on(copy_events[i]);
115 h.parallel_for(sycl::nd_range<1>(a_tile.shape()[0] * wg, wg),
116 [=](auto item) {
117 auto row_index = item.get_group(0);
118 auto local_id = item.get_local_id();
119 auto group_size = item.get_local_range(0);
120
121 auto row = a_tile.row(row_index);
122
123 for (std::size_t idx = local_id; idx < row.size();
124 idx += group_size) {
125 auto &&[index, a_v] = row[idx];
126 auto &&[i, k] = index;
127
128 auto &&b_v = *(b_iter + k);
129 auto &&c_v = *(c_iter + i);
130
131 c_v += a_v * b_v;
132 }
133 });
134 });
135 comp_events.push_back(event);
136 }
137
138 __detail::wait(comp_events);
139}
