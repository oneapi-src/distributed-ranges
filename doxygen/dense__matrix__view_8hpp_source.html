132 size_type size() const noexcept { return shape()[0] * shape()[1]; }
133
134 scalar_reference operator[](key_type idx) const {
135 return data_[idx[0] * ld_ + idx[1]];
136 }
137
138 iterator begin() const {
139 return iterator(data_, key_type{0, 0}, idx_offset_, shape_, ld_);
140 }
141
142 iterator end() const {
143 return iterator(data_, key_type{shape_[0], 0}, idx_offset_, shape_, ld_);
144 }

146 auto row(size_type row_index) const {
147 // return dense_matrix_row_view(data_ + row_index * ld_, row_index,
148 // shape()[1]);
149 auto row_elements = rng::views::iota(size_type(0), size_type(shape()[1]));
150 Iter data = data_ + row_index * ld_;
151
152 return row_elements | rng::views::transform([=](auto column_index) {
153 return reference(key_type(row_index, column_index),
154 data[column_index]);
155 });
156 }
157
158 auto column(size_type column_index) const {
159 // return dense_matrix_column_view(data_ + column_index, column_index,
160 // shape()[0], ld_);
161 auto column_elements =
162 rng::views::iota(size_type(0), size_type(shape()[0]));
163 Iter data = data_ + column_index;
164 size_type ld = ld_;
165
166 return column_elements | rng::views::transform([=](auto row_index) {
167 return reference(key_type(row_index, column_index),
168 data[row_index * ld]);
169 });
170 }
171
172 iterator data() const { return data_; }
173
174 std::size_t rank() const { return rank_; }
175
176private:

178 key_type shape_;
179 key_type idx_offset_;
180 size_type ld_;
181 size_type rank_;
